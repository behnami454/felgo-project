Creating Grid for this page (which you can check the all details here Felgo Grid )

Grid {
    id: grid
    width: 400
    height: 400
    anchors.verticalCenter: parent.verticalCenter
    anchors.verticalCenterOffset: 42
    anchors.horizontalCenterOffset: 80
    anchors.horizontalCenter: parent.horizontalCenter
    z: 10
    rotation: 0
    rows: 4
    state: ""
    clip: false
    spacing: 38
    columns: 2
  }
